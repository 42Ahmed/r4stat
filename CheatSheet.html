Creating functions#

avg  <- function(x){
    s <- sum(x)
    n  <-  length(x)
    s/n
}

x  <- 1:100

avg(x)
50.5
mean(x)
50.5
identical(mean(x),avg(x))
TRUE
remainder <- function(num, divisor = 2) {
  # Remember: the last expression evaluated will be returned! 
  num %% divisor
}

remainder(7)
1
remainder(8,3)
2
evaluate <- function(func, data){
func(data)
}
